```mermaid
graph TB
  %% ノード定義
  start(["START"])
  node_1("①引数チェック")
  fork_1{"甲が存在するか"}
  node_2("②甲を削除")  
  node_3("③リストの初期化")  
  node_4("④乙を取得")
  subgraph loop_1[" "]
    loop_1_start[/"ループA開始"\]
    loop_1_goal[\"ループA終了"/]
    fork_2{"丙が存在するか"}
    subgraph loop_2[" "]
      loop_2_start[/"ループB開始"\]
      loop_2_goal[\"ループB終了"/]  
      node_5("⑤丁を取得")
      node_6("⑥丁をリストに追加")        
    end
  end
  node_7("⑦甲の作成")
  node_8("⑧リストを甲に出力")  
  error_1("エラー処理")    
  goal(["END"])

  %% ノードスタイル定義
  style loop_1 fill:#ffa23e
  style loop_2 fill:#7BCCAC

  %% リンク定義

  node_1 -. "引数不正" .-> error_1  
  node_1 --> fork_1
  fork_1 --> |"存在する"| node_2   
  node_2 -. "削除失敗" .-> error_1  
  node_2 --> node_3
  fork_1 --> |"存在しない"| node_3
  node_3 --> node_4
  node_4 -. "取得失敗" .-> error_1  
  node_4 --> loop_1_start
  loop_1_start --> fork_2
  fork_2 --> |"存在する"| loop_2_start
  loop_2_start --> node_5
  node_5 -. "取得失敗" .-> error_1
  node_5 --> node_6
  node_6 --> loop_2_goal
  loop_2_goal --> loop_1_goal
  fork_2 --> |"存在しない"| loop_1_goal
  loop_1_goal --> node_7
  node_7 -. "作成失敗" .-> error_1  
  node_7 --> node_8
  node_8 -. "出力失敗" .-> error_1  
  node_8 --> goal  
  error_1 --> goal
```
